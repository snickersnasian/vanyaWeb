clear - очистить консоль
touch <название файла>
cd .. - выйти из папки

git init - включить сохранение
git add . - добавить все файлы для схранения
git status - посомтреть, что будет сохраняться
git commit -m "Тут твое сообщение" - сделать сохранение
git remote add origin <тут твой ssh> - добавить папку сохранения в интеренете
git push origin - загрузить сохранение в папку в интернете
git clone <тут твой ssh> - скачать репозиторий к себе
git pull origin - сохранить к себе изменения

code . - запустить VS Code
